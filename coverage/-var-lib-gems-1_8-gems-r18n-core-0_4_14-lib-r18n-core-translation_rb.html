27   # Translation is container of translated messages.
 
28   #
 
29   # You can load several locales and if translation willn’t be found in first,
 
30   # r18n will be search it in next. Use R18n::I18n.new to load translations.
 
31   #
 
32   # To get translation value use method with same name. If translation name
 
33   # is equal with Object methods (+new+, +to_s+, +methods+) use
 
34   # <tt>[name, params…]</tt>. If you want to get pluralizable value, just set
 
35   # value for pluralization in first argument of method. See samples below.
 
36   #
 
37   # Translated strings will have +locale+ methods, which return Locale or
 
38   # UnsupportedLocale, if locale file isn’t exists.

75     # This is internal a constructor. To load translation use

77     def initialize(main_locale, path = '', locale = nil, data = {})
 
78       @path = path
 
79       @data = {}
 
80       @locale = main_locale
 
81       merge! data, locale unless data.empty?
 
82     end

84     # Add another hash with +translations+ for some +locale+. Current data is
 
85     # more priority, that new one in +translations+.
 
86     def merge!(translations, locale)
 
87       translations.each_pair do |name, value|
 
88         if not @data.has_key? name
 
89           path = @path.empty? ? name : "#{@path}.#{name}"
 
90           case value
 
91           when Hash
 
92             value = Translation.new(@locale, path, locale, value)
 
93           when String
 
94             c = { :locale => locale, :path => path }
 
95             v = Filters.process_string(Filters.passive_enabled, value, c, [])
 
96             value = TranslatedString.new(v, locale, path)
 
97           when Typed
 
98             value.locale = locale
 
99             value.path = path
 
100             unless Filters.passive_enabled[value.type].empty?
 
101               value = Filters.process(Filters.passive_enabled, value.type,
 
102                                       value.value, value.locale, value.path, {})
 
103             end
 
104           end
 
105           @data[name] = value
 
106         elsif @data[name].is_a? Translation
 
107           @data[name].merge! value, locale
 
108         end
 
109       end
 
110     end
