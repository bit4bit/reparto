24 module R18n
 
25   # Information about locale (language, country and other special variant
 
26   # preferences). Locale was named by RFC 3066. For example, locale for French
 
27   # speaking people in Canada will be +fr-CA+.
 
28   #
 
29   # Locale classes are placed in <tt>R18n::Locales</tt> module and storage
 
30   # install <tt>locales/</tt> dir.
 
31   #
 
32   # Each locale has +sublocales+ – often known languages for people from this
 
33   # locale. For example, many Belorussians know Russian and English. If there
 
34   # is’t translation for Belorussian, it will be searched in Russian and next in
 
35   # English translations.
 
36   #
 
37   # == Usage
 
38   #
 
39   # Get Russian locale and print it information
 
40   #
 
41   #   ru = R18n::Locale.load('ru')
 
42   #   ru.title #=> "Русский"
 
43   #   ru.code  #=> "ru"
 
44   #   ru.ltr?  #=> true
 
45   #
 
46   # == Available data
 
47   #
 
48   # * +code+ – locale RFC 3066 code;
 
49   # * +title+ – locale name on it language;
 
50   # * +ltr?+ – true on left-to-right writing direction, false for Arabic and
 
51   #   Hebrew);
 
52   # * +sublocales+ – often known languages for people from this locale;
 
53   # * +week_start+ – does week start from +:monday+ or +:sunday+.
 
54   #
 
55   # You can see more available data about locale in samples in
 
56   # <tt>locales/</tt> dir.

75     def self.load(code)
 
76       original = code.to_s.gsub(/[^-_a-zA-Z]/, '')
 
77       code = original.gsub('_', '-').downcase
 
78 
79       @@loaded[code] ||= begin
 
80         if exists? code

82           name = code.gsub(/[\w\d]+/) { |i| i.capitalize }.gsub('-', '')

141     # Convert +object+ to String. It support Fixnum, Bignum, Float, Time, Date
 
142     # and DateTime.
 
143     #
 
144     # For time classes you can set +format+ in standard +strftime+ form,
 
145     # <tt>:full</tt> (“01 Jule, 2009”), <tt>:human</tt> (“yesterday”),
 
146     # <tt>:standard</tt> (“07/01/09”) or <tt>:month</tt> for standalone month
 
147     # name. Default format is <tt>:standard</tt>.
 
148     def localize(obj, format = nil, *params)
 
149       case obj
 
150       when Integer
 
151         format_integer(obj)
 
152       when Float
 
153         format_float(obj)
 
154       when Time, DateTime, Date
 
155         return strftime(obj, format) if format.is_a? String
 
156         return month_standalone[obj.month - 1] if :month == format
 
157         return obj.to_s if :human == format and not params.first.is_a? I18n
 
158 
159         type = obj.is_a?(Date) ? 'date' : 'time'
 
160         format = :standard unless format
 
161 
162         unless [:human, :full, :standard].include? format
 
163           raise ArgumentError, "Unknown time formatter #{format}"
 
164         end
 
165 
166         send "format_#{type}_#{format}", obj, *params
 
167       else
 
168         obj.to_s
 
169       end
 
170     end

191     # Same that <tt>Time.strftime</tt>, but translate months and week days
 
192     # names. In +time+ you can use Time, DateTime or Date object. In +format+
 
193     # you can use standard +strftime+ format.
 
194     def strftime(time, format)
 
195       translated = ''
 
196       format.scan(/%[EO]?.|./o) do |c|
 
197         case c.sub(/^%[EO]?(.)$/o, '%\\1')
 
198         when '%A'
 
199           translated << wday_names[time.wday]
 
200         when '%a'
 
201           translated << wday_abbrs[time.wday]
 
202         when '%B'
 
203           translated << month_names[time.month - 1]
 
204         when '%b'
 
205           translated << month_abbrs[time.month - 1]
 
206         when '%p'
 
207           translated << (time.hour < 12 ? time_am : time_pm)
 
208         else
 
209           translated << c
 
210         end
 
211       end
 
212       time.strftime(translated)
 
213     end

220     # Format +time+ in human usable form. For example “5 minutes ago” or
 
221     # “yesterday”. In +now+ you can set base time, which be used to get relative
 
222     # time. For special cases you can replace it in locale’s class.
 
223     def format_time_human(time, i18n, now = Time.now, *params)
 
224       minutes = (time - now) / 60.0
 
225       diff = minutes.abs
 
226       if (diff > 24 * 60) or (time.mday != now.mday and diff > 12 * 24)
 
227         format_date_human(R18n::Utils.to_date(time), i18n,
 
228                           R18n::Utils.to_date(now)) + format_time(time)
 
229       else
 
230         if -1 < minutes and 1 > minutes
 
231           i18n.human_time.now
 
232         elsif 60 <= minutes
 
233           i18n.human_time.after_hours((diff / 60.0).floor)
 
234         elsif -60 >= minutes
 
235           i18n.human_time.hours_ago((diff / 60.0).floor)
 
236         elsif 0 < minutes
 
237           i18n.human_time.after_minutes(minutes.round)
 
238         else
 
239           i18n.human_time.minutes_ago(minutes.round.abs)
 
240         end
 
241       end
 
242     end
